  if (errorMessage.includes('net::ERR_ABORTED')) {
    // Handle redirects/blocked content
    const currentUrl = page.url();
    if (currentUrl !== url) {
      // Page redirected successfully
      return page;
    }

  // Retry with more lenient settings

}
